In [86]:
from sklearn import metrics

def accuracy(ans, predicted):
    sum = 0
    n = len(predicted)
    count= 0
    for y in ans:
        temp = 0
        if predicted[count] < 0.5:
            temp = 0
        else:
            temp = 1
        if temp == y:
            sum += 1
        count += 1
    return sum / n
